class Solution {
    public int majorityElement(int[] nums) {
      int count,e;
       e=nums[0];
        count=1;
        for(int i=1;i<nums.length;i++){
            if(nums[i]!=e) count--;
            else count++;
            if(count==0){
                e=nums[i];
                count=1;
            }
            if(count>nums.length/2) return e;
        }
        return e;
    }
}
