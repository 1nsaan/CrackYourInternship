/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        ListNode la = new ListNode();
        ListNode curr = headA;
        int lena=0,lenb=0;
        while(curr!=null){
                 curr=curr.next;
                 lena++;
        }
        ListNode lb = new ListNode();
        curr = headB;
        
        while(curr!=null){
             curr=curr.next;
             lenb++;
        }

        while(lena>lenb){
            headA = headA.next;
            lena--;
        }

        while(lenb>lena){
            headB = headB.next;
            lenb--;
        }
     
        while(headA!=null && headB!=null){
            //System.out.println(la.val + " " + lb.val);
            if(headA==headB){
                return headA;
             }
            headA = headA.next;
            headB = headB.next; 
        }

        return null;
    }
}
