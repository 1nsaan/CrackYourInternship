class Solution {
    public int strStr(String haystack, String needle) {
        int i=0,j=0;
        while(i<haystack.length() && j<needle.length()){
            if(i+j< haystack.length() && haystack.charAt(i+j)==needle.charAt(j)) j++;
            else{
                j=0;
                i++;
            }
            if(j==needle.length()) return i;
        }
         if(j==needle.length()) return i;
        return -1;
    }
}
